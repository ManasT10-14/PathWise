```python
def write_analysis_memory(user_id: str, chain_result: dict, roadmap_id: str) -> None:
    """Append analysis snapshot to users/{uid}/learner_memory/analysis_history.
    
    Reads existing doc, appends new entry (max 10 kept, drop oldest), rewrites.
    Entry shape: {"roadmap_id": str, "target_role": str, "gap_count": int, 
                  "phase_count": int, "analyzed_at": SERVER_TIMESTAMP_as_str (use isoformat)}
    Use firestore.SERVER_TIMESTAMP for analyzed_at field in the Firestore doc.
    Uses db.collection("users").document(user_id).collection("learner_memory").document("analysis_history").
    On first call the doc won't exist — use .set() with merge=False (just set fresh).
    """

def write_expert_annotation(user_id: str, roadmap_id: str, milestone_level: str, 
                             annotation_text: str, expert_id: str) -> None:
    """Append expert annotation to users/{uid}/learner_memory/expert_annotations.
    
    Entry shape: {"roadmap_id": str, "milestone_level": str, "annotation": str, 
                  "expert_id": str, "annotated_at": SERVER_TIMESTAMP}
    Append to existing "annotations" list field using ArrayUnion, or set if doc missing.
    Use db.collection("users").document(user_id).collection("learner_memory").document("expert_annotations").
    """

def read_learner_memory(user_id: str) -> dict:
    """Read all learner_memory subdocs and return a merged dict for prompt injection.
    
    Reads: analysis_history, expert_annotations
    Returns:
    {
        "analysis_history": list[dict],   # from analysis_history.entries or []
        "expert_annotations": list[dict], # from expert_annotations.annotations or []
        "has_memory": bool                # True if either list is non-empty
    }
    Use db.collection("users").document(user_id).collection("learner_memory") batch get.
    Never raise — return empty dict structure on any error (memory is enrichment, not required).
    """
```

All three functions use `db = firestore.client()` at call time (same pattern as firestore_writer.py).

```python
"""Replan chain — memory-aware single-step roadmap adjustment.

Unlike the 4-step analysis chain, replanning is a single Gemini call that
uses the existing roadmap + progress + memory as input context.
"""
import time
from pathlib import Path

import structlog
from services.gemini_client import get_client   # import pattern from existing services

log = structlog.get_logger(__name__)

REPLANNER_PROMPT = (Path(__file__).parent.parent / "prompts" / "replanner.txt").read_text()


class ReplanStepError(Exception):
    def __init__(self, cause: Exception) -> None:
        self.cause = cause
        super().__init__(f"Replan failed: {cause}")


def run_replan_chain(
    roadmap_id: str,
    target_role: str,
    current_milestones: list[str],
    current_timeline: str,
    stage_progress: dict[str, float],
    learner_feedback: str,
    memory: dict,           # from read_learner_memory()
    stall_days: int | None, # None if user-triggered (not stall-triggered)
) -> dict:
    """Run the replan prompt chain.
    
    Returns dict with keys:
      "adjusted_milestones": list[str]    — new milestone strings (legacy format)
      "replan_reason": str                — AI explanation of changes
      "stalled_stages": list[str]         — stages identified as stalled
    
    Raises ReplanStepError on Gemini failure.
    """
```

Build the prompt by formatting REPLANNER_PROMPT with:
- `progress_summary`: `", ".join(f"{k}: {v:.1f}" for k,v in stage_progress.items())`
- `stall_context`: `f"Stage '{stalled}' unchanged for {stall_days} days (threshold: 14 days)"` if stall_days else `"User-triggered replan (no automatic stall detected)"`
- `memory_context`: format `memory["analysis_history"]` as `"- {r['target_role']}, {r['gap_count']} gaps, analyzed_at: {r.get('analyzed_at','?')}"` lines, or `"No prior session history"`
- `expert_annotations`: format `memory["expert_annotations"]` as `"- [{r['milestone_level']}] {r['annotation']}"` lines, or `"No expert annotations"`
